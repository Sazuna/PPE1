function getScrollTop(){  
	var scrollTop=0;
	if(document.documentElement&&document.documentElement.scrollTop){  
		scrollTop=document.documentElement.scrollTop;
	}else if(document.body){  
		scrollTop=document.body.scrollTop;  
	}  
	return scrollTop;
}

function getClientHeight(){  
	var clientHeight=0;  
	if(document.body.clientHeight&&document.documentElement.clientHeight){  
		var clientHeight = (document.body.clientHeight<document.documentElement.clientHeight)?document.body.clientHeight:document.documentElement.clientHeight;
	}else  {  
		var clientHeight = (document.body.clientHeight>document.documentElement.clientHeight)?document.body.clientHeight:document.documentElement.clientHeight;      
	}  

	return clientHeight;  
}  

function getScrollHeight(){  
	return Math.max(document.body.scrollHeight,document.documentElement.scrollHeight);  
}

function clone(myObj){
	if(typeof(myObj) != 'object') return myObj;
	if(myObj == null) return myObj;
	var myNewObj = new Object();
	for(var i in myObj)
		myNewObj[i] = myObj[i];//clone(myObj[i]);
	return myNewObj;
}

<p>　　<strong>女人相争 宫斗影子随处可见</strong></p>
<p>　　《爱情悠悠药草香》播出近一周的时间里，不断有评论称该剧就是一部民国版的《甄执》。剧中妻妾之间为争宠上位而使劲手段，为了寻得富贵的生活想尽一切办法怀孕，这种思路几乎就是宫廷女人的进阶史。下药、流产、扮鬼、找替罪羊，一切恶行都是为了保住自己的地位，而这些就更是宫廷女人才会做的事。尤其是宣萱饰演的二爷的大太太，名正言顺地打压着所有会对自己造成威胁的人。有网友称：“这俨然一副皇后样，背后使坏却让人浑然不知。”</p>
<p>　　虽然宣萱的角色属于腹黑型，但她却有别于所谓的“皇后”，民国时期虽然是一夫一妻多妾，但妻子在家中的地位还是无法被轻易撼动，再加上家族间存在着利益关系，所以夫妻二人还是处于相对平等的地位，并非与宫廷一样。而宣萱所展现的绝对权力，不只是针对妾室，就连与丈夫白乾笙（韩栋饰）也会有现代夫妻之间的争吵拌嘴，《爱情悠悠药草香》是一部更接近于现代人的宅斗大戏。比起宣萱的腹黑程度，倪虹洁饰演的二姨太也是个不折不扣的“幕后黑手”，暗中挑拨离间实则为了死去的儿子复仇，谈到这个角色的悲情命运，倪虹洁本人表示：“一切都是为了争宠，我总结暗争是一种狡猾的手段，但是明斗就会死的早。”导演也称：“《爱情悠悠药草香》女人之间极致的矛盾很吸引我，这部年代戏有自己的味道。”</p>
<p>　　<strong>励志传奇 堪称民国版大长今</strong></p>
<p>　　《爱情悠悠药草香》不仅是一部豪门宅斗戏，更是一部讲述了励志人生的正能量情感大戏。女主角娄艺潇饰演的黄采薇，是个从仆人之女摇身一变成为受宠姨太太的女子，她精通药理，懂得经营之道，更会为人处事。尽管生活的身不由己，但在这样一个狭窄的宅院内生存，她深谙豪门之内的规则。虽然无数次被虐被陷害，但一颗打不倒的坚强内心还是支撑着她生存了下来。有很多观众表示：“这不就是当年宫廷里的大长今吗！什么都懂什么都会，还一直很善良。”</p>
<p>　　娄艺潇在接受采访时也表示：“黄采薇身上有一种恬静又大气的双重性格，对待任何人也是友善为先。”无辜被虐的她总是希望能够平息事态，不要连累更多的人，但命运弄人，因为周转在兄弟二人的感情中间，采薇不断遭受到闲言碎语，仿佛怎么做怎么都是错。一直在委屈自己成全别人的采薇，也让观众爱上了这个白家大院内的女人。</p>
<p>　　<strong>唯美爱情 开启琼瑶式的浪漫</strong></p>
<p>　　辛迪加影视公司专注于出品美人、美景、美情的“爱情美剧”，尽管《爱情悠悠药草香》是一部年代戏，但是在爱情的唯美程度上绝对不输浪漫之母琼瑶阿姨。从拍摄的角度说，导演蒋家骏有着执导多部爱情戏的经验基础；从剧本的角度说，豪宅之内争斗不断，却始终存在着善良的人，这就与阴谋算计形成了一种中和；从爱情的角度说，表面的一夫多妻都只是特殊年代的无奈选择，在男女主角心中仍然有美好单纯的爱。韩栋深知自己心爱的女人爱着自己的大哥，但经过多年的等待，终于在道德舆论允许的情况下娶回爱妻，用真心实意打动了这个固执于过去的女子。有观众调侃说：“这么浪漫的故事只有琼瑶阿姨的剧本里才会有吧！“</p>
<p>　　有网友把《爱情悠悠药草香》与琼瑶式爱情作比：“男女主角日久生情，一方出身低贱无背景，一方高贵在上有学识，他们的爱情遭到家庭的反对，后因二人坚持不懈，用真心证明了爱情可以冲破一切，终于赢得了圆满结为秦晋之好。”剧中，韩栋与娄艺潇从小一起长大，有着相同的人生目标和兴趣爱好，对任何人都心存善意，惺惺相惜的二人会通过自己的行动为观众撒下爱。</p>
<p>　　<strong>设计陷害 充满侦探剧的悬疑</strong></p>
<p>　　《爱情悠悠药草香》一开篇便是三姨太仝晓燕被陷害导致两度流产，无论是太太指使下人在三姨太的鞋上动手脚制造意外的假象，还是半夜扮鬼导致三姨太惊吓过度流产，种种手段之阴险让观众直呼：“这简直就是悬疑作案的节奏啊！”擅长女人戏的导演蒋家骏透露：“剧中细腻的女人争斗让我觉得很有成就感，除此之外，女人使坏的&lsquo;高招&rsquo;也是层出不穷。”除了女人之间的暗斗，生意场上的设计陷害也一直存在，马馥芳的父亲马国安（曾江饰）堪称商场上的“老油条”，为了获得更多的利益，他不惜设圈套杀害自己的女婿白乾笙，令人发指的行为也为该剧增添了更多的精彩元素。</p>
<p>　　只要宅院内有姨太太怀孕或是得宠，总会发生一些惨剧，看上去真实的像是意外现场，但实际上都是人为操纵。精彩的剧情配合着迷雾缭绕的音效，漆黑的夜晚总要发生一些惊人的事件。而《爱情悠悠药草香》所讲述的药草世家白家大院，是以经营药品而闻名于当时，剧中常常出现的“三七”也成了治愈神药，所有的意外只要有“三七”，定会有立竿见影的效果，网友调侃：“真是侦探悬疑外带治愈的一条龙服务啊。”</p>
<p>　　<strong>家庭纠纷 婆媳矛盾亘古不变</strong></p>
<p>　　《爱情悠悠药草香》既然是发生在豪宅内，又有这么多的儿媳妇，剧中自然少不了亘古不变的婆媳矛盾。白老太太（夏台凤饰）有着绝对的当家地位，但出于亲生儿子在家中的地位还未稳固，所以百般阻碍二少爷白乾笙以及多房妻妾的发展。面对面的质问惩戒，背地里的暗插奸细，白家的大小事情都在她的掌控范围之下。虽然儿媳妇们心知肚明，但却无人敢反抗，封建残余制度下的婆媳之争还是以唯听是从为主。</p>
<p>　　《爱情悠悠药草香》集合了所有与家庭、女人、夫妻有关的看点，虽然看似是一部杂烩剧，但剧中所诠释的人性和家庭核心还是让很多观众看到了一部浓缩的民国史，其中蕴藏着观众或了解或不了解的多面内容。在女人争斗中也暗示了用人和生存的智慧，《爱情悠悠药草香》是值得用心品看的爱情励志大戏。</p>
