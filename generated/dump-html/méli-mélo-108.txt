    <h1 class="page-header">Courgettes farcies au méli-mélo céréales et légumes secs Tipiak, potiron &amp; bacon</h1>

                                                </div>
              <div class="p-4">
                <div class="text-gold uppercase text-center mb-2" style="font-size:1.2em;">Ingrédients</div>
                <div style="font-size:0.8em;text-justify: inter-word;">
                  1 sachet de Méli-Mélo Céréales et Légumes secs Tipiak 8 courgettes rondes 2 tomates 250 g de potiron 1 cuillère soupe d’huile d’olive 1 oignon 5 tranches de bacon 20 olives noires dénoyautées coupées 2 cuillères à soupe de persil plat et de coriandre
                </div>
              </div>
            </div>
            <div class="col-12 col-md-6 text-light">
              <div class="p-2">
                <big class="text-gold uppercase" style="font-size:1.2em;">Préparation</big><br/>
                <div style="font-size:0.9em;text-justify: inter-word;">
                  1/ Portez 30 cl d’eau à ébullition dans une poêle. Hors du feu, versez le contenu d’un sachet de Méli-Mélo Céréales et Légumes secs Tipiak et ajoutez une noix de beurre. Mélangez, couvrez et laissez gonfler 9 minutes. Réservez.	2/ Coupez la partie supérieure des courgettes et le potiron en tranches puis cuisez-les 10 minutes à la vapeur.	3/ Videz l’intérieur des courgettes et coupez en dés le potiron cuit et les tomates, puis mettez de côté dans un saladier.	4/ Dans une poêle, faites revenir l’oignon émincé dans l’huile d’olive puis ajoutez les tranches de bacon coupées en lanières, la chair des courgettes, les tomates et les dés de potiron. Faites cuire jusqu’à absorption totale du jus des légumes.	5/ Ajoutez en fin de préparation les olives, le persil et la coriandre hachés ainsi que la préparation Méli-Mélo Céréales et Légumes secs Tipiak.	6/ Réchauffez le tout pendant 2 à 3 minutes, puis farcissez les courgettes et enfournez pour 10 minutes à 160°C.
                </div>

      function convertNoteIntoStars(note) {
        var htmlToReturn = '<span class="font-size-0_7">';
        if (note == null || note == 0) {
          return '';
        }
        if (note < 20) {
          htmlToReturn = '<i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i>';
        }
        if (note == 1) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i>';
        }
        if (note == 2) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="fas fa-star"></i><i class="far fa-star"></i>';
        }
        if (note == 3) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="fas fa-star"></i><i class="fas fa-star"></i>';
        }
        if (note >= 20) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i>';
        }
        if (note >= 40) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="fas fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i>';
        }
        if (note >= 60) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="fas fa-star"></i><i class="fas fa-star"></i><i class="far fa-star"></i><i class="far fa-star"></i>';
        }
        if (note >= 80) {
          htmlToReturn = '<i class="fas fa-star"></i><i class="fas fa-star"></i><i class="fas fa-star"></i><i class="fas fa-star"></i><i class="far fa-star"></i>';

        }
        htmlToReturn += '</span><br/>';
        return htmlToReturn;
      }

            limit: 3, // Nombre maximum de résultats à afficher
            published: 1 // TRUE si on veut que les éléments publiés
          };
          $.ajax({
            url: "/membres-search",
            type: 'get',
            dataType: 'json',
            beforeSend: function () {
              $('#search_engine_picto').html('<i class="fas fa-spinner fa-pulse"></i>');
            },
            success: function (data) {
              $('#search_engine_picto').html('<i class="fab fa-searchengin"></i>');
              $('#header_search_results').show();
              $('#header_search_results').removeClass('d-none');
              var results_html = '';
              var link_path = "/resultats-de-recherche?search%5Bquery%5D=termes&amp;type=chefs";
              link_path = link_path.replace('termes', $('#search_engine').val());
              results_html = results_html + '<hr class="bg-light my-1">';
              results_html = results_html + '<div class="autocomplete_categorie">Chefs (<a title="Voir tous les chefs" href="' + link_path + '"><i>Voir tous les chefs</i></a>)</div>';
              results_html += '<div class="row">';
              $.each(data.data.membres, function (index, element) {
                results_html += '<div class="col-12 col-md-6 col-lg-4">';
                if (element.filename != null) {
                  results_html += '<a title="' + element.titre + '" class="d-block p-2 text-light" href="/membres/' + element.slug + '">';
                  results_html += '<div class="row">';
                  results_html += '<div class="col-12 col-sm-3 col-lg-4 d-none d-sm-block">';
                  results_html += '<div class="block-img-search-bar">';
                  results_html += '<img class="img-fluid d-none d-sm-block img-search-bar" src="/uploads/img/' + element.filename + '" alt="' + element.alt + '" />';
                  results_html += '</div>';
                  results_html += '</div>';
                  results_html += '<div class="col-12 col-sm-9 col-lg-8">';
                  results_html += '<span class="text-gold uppercase italic font-size-0_9">' + element.titre + '</span><br/>';
                  results_html += convertNoteIntoStars(element.etoiles);
                  results_html += '<div class"font-size-0_7" style="line-height:1.1;">' + element.contenu + '</div>';
                  results_html += '</div>';
                  results_html += '</div>';
                  results_html += '</a>';
                } else {
                  results_html += '<a title="' + element.titre + '" href="/tests-produit/' + element.slug + '">';
                  results_html += element.titre;
                  results_html += '</a>';
                }
                results_html += '</div>';
              });
              results_html += '</div>';
              $('#header_search_membres').html(results_html);
            },
            data: parametres
          });
        }
      }
    </script>
