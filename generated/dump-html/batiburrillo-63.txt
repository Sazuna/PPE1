var triggerAuth = function(opts) {
	try {
		opts = opts || {};
		opts.webtypesAllowed = ["story-reportage", "story-video", "la-contra", "lacontra", "story", "opinion"];
		var sections = gg.paywall.utils.getSections(ggPageData.sections);
		var category = gg.paywall.utils.getCategory(ggPageData.sections);
		var accessType = gg.paywall.utils.getAccessType(ggPageData.accessType);
		window.params = window["params"] || {};	

		params["articleId"] = ggPageData.contentId;				
		params["section"] = sections[0];				
		params["category"] = category;		   
		params["webtype"] = ggPageData.webType;					
		params["authors"] = gg.paywall.utils.removeHash(ggPageData.authorName);						
		params["site"] = window.location.hostname;		

		if (typeof adsCfg !== "undefined") {
			if (typeof adsCfg.keys['topics'] !== "undefined") {
				params["tags"] = Array.isArray(adsCfg.keys['topics']) ? adsCfg.keys['topics'].join(',') : adsCfg.keys['topics'];				

			}
		}
		
		params["publicationDate"] = gg.paywall.utils.getDate(ggPageData.publicationDate);
		params["modificationDate"] = gg.paywall.utils.getDate(ggPageData.modificationDate);					
		params["facet"]= ggPageData.facet;					
		params["url1"]= sections[0];		
		params["url2"]= category.split("/")[0]; 	
		params["accessType"] = accessType;	
		params["platform"] = opts.isApp ? (gg.accounts.getCustomUserAgent() || "").replace(/[^A-Za-z]/g, "") : ggPageData.facet;

		if (gg.accounts.didomiLogged() || (Didomi && Didomi.getUserConsentStatusForVendor('c:evolok-izKz7AYV'))) {
			if (!opts.isInfiniteScroll || !opts.onAuthTriggered || (opts.isApp && !gg.accounts.didomiLogged())) {
				if (opts.onAuth && typeof opts.onAuth === "function") {
					opts.onAuth(params, opts.token, opts.platform);
					opts.onAuthTriggered = true;
				}
			} else {
				gg.paywall.verifyAuthorization(params, opts.isApp);
			}		
		} else {
			if (accessType === "yes" && (opts.onNoConsent && typeof opts.onNoConsent === "function")) {
				opts.onNoConsent(params, opts);
				return;
			}
			if (!opts.isApp && opts.dab && opts.dab.active) {
				gg.paywall.triggerDabDetection(params);
			}	
		}
	} catch(e) {
		console.log(e);
	}
};

var registerParams = function() {
	if (__evContext.isParamsHandlerRegistered) {
		return;
	}

	window.didomiOnReady = window.didomiOnReady || [];
	window.didomiOnReady.push(function (Didomi) {
		try {
			__evContext.articleContainer = document.querySelectorAll("article.article-default")[0];

			if (window.ggReady || (document.readyState === 'complete')) {
				gg.didomi.onTCLoaded(function(){
            		triggerAuth(__evContext);
        		});
				return;
			}
			document.addEventListener('readystatechange', (e) => {
				if (document.readyState === 'complete') {
					gg.didomi.onTCLoaded(function(){
            			triggerAuth(__evContext);
        			});
				}
			});
			__evContext.isParamsHandlerRegistered = true;
		} catch(e) {
			console.log(e);
		}
	});
};

var registerInfiniteScroll = function() {
	if (__evContext.isInfiniteScrollHandlerRegistered) {
		return;
	}
	document.addEventListener("infiniteScrollChangeUrl", function(event) {
		if (!__evContext.isInfiniteScroll) {
			__evContext.isInfiniteScroll = true;
		}
		__evContext.articleContainer = event.articleContainer;
		triggerAuth(__evContext);	
	});
	__evContext.isInfiniteScrollHandlerRegistered = true;
};

                        <p data-gtm-element-container="modulo-texto-link" class="paragraph">Fatiga intensa, trastornos del sueño y atención, un dolor crónico y persistente... todos estos son síntomas que experimentan los pacientes que padecen <strong>fibromialgia</strong>. Todo este batiburrillo de síntomas se traduce en una <strong>reducción de la capacidad para realizar las actividades de la vida cotidiana</strong>, por lo que muchos pacientes acaban sufriendo <strong>ansiedad</strong>. Según el estudio EPISER de la Sociedad Española de Reumatología, el 2,4% de la población adulta española, lo que supone cerca de 900.000 personas afectadas, tiene que lidiar con esta situación. </p>
<span class="content-ad-module ">
    
    
    
<span class="content-ad visible-xs roba-container  hidden-ad"><span class="ad-unit roba">
                <ins data-ad-type="roba" data-ad-device="xs" data-ad-sizes='["fluid",[336,280],[300,250],[300,300],[300,600]]' data-ad-keys="{}" data-ad-lazy="false" data-event-scroll-xs="" data-name="roba"></ins>
                </span></span><span class="content-ad visible-md irr-container  hidden-ad"><span class="ad-unit irr">
                <ins data-ad-type="irr" data-ad-device="md" data-ad-keys="{}" data-ad-lazy="false"></ins>
                </span></span><span class="content-html"><span class="intext-md hidden-xs hidden-md" w-constraint> </span></span></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">Por el momento, se desconoce la causa exacta de esta enfermedad crónica. De esta manera, <strong>no existe un tratamiento específico </strong>para curarla. Para mejorar la calidad de vida de los pacientes, los médicos pueden aconsejar atención psicológica si es necesario, rehabilitación física o <strong>analgésicos </strong>como tratamiento para calmar los dolores. </p>
<span class="content-ad-module ">
    
    
    
<span class="content-ad visible-xs irr-container  hidden-ad"><span class="ad-unit irr">
                <ins data-ad-type="irr" data-ad-device="xs" data-ad-keys="{}" data-ad-lazy="false"></ins>
                </span></span><span class="content-html"><span class="intext-xs hidden-xs hidden-md" w-constraint></span></span></span><span class="content-subtitle-simple ">

    

<h2 class="subtitle">Reducen la ansiedad cuatro veces más</h2></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">Aparte de estos tratamientos, existen<strong> pocas alternativas</strong> para combatir la fibromialgia. Pero esto puede ser por poco tiempo a la vista de los resultados de un <a href="https://www.asahq.org/about-asa/newsroom/news-releases/2022/10/green-eyeglasses-reduce-pain-related-anxiety-in-fibromyalgia-patients-study-shows#:~:text=NEW%20ORLEANS%20%E2%80%94%20Wearing%20special%20green,ANESTHESIOLOGY%C2%AE%202022%20annual%20meeting." rel="nofollow noreferrer">estudio </a>presentado en la reunión anual de la <a href="https://nam10.safelinks.protection.outlook.com/?url=https://www.asahq.org/annualmeeting&amp;data=05%7C01%7Cdmills@healthline.com%7Cd51d8fcfe97b4df867e908daabbfc136%7C00e1df3d9626410c898c16aaa8c2afc9%7C0%7C0%7C638011137310968879%7CUnknown%7CTWFpbGZsb3d8eyJWIjoiMC4wLjAwMDAiLCJQIjoiV2luMzIiLCJBTiI6Ik1haWwiLCJXVCI6Mn0=%7C3000%7C%7C%7C&amp;sdata=tPsZPiP+NzUJ1MRmvsz+qMGRA9XJ/kggsB1U/iuRQ1w=&amp;reserved=0" rel="nofollow noreferrer">Sociedad Americana de Anestesiólogos</a>. Al parecer, una posible solución para lidiar con la<strong> ansiedad que provocan los dolores</strong> de la fibromialgia en los pacientes es el uso de una<strong>s gafas con cristales verdes especiales</strong>.</p>
<span class="content-ad-module ">
    
    
    
<span class="content-ad visible-xs roba-container  hidden-ad"><span class="ad-unit roba">
                <ins data-ad-type="roba" data-ad-device="xs" data-ad-sizes='["fluid",[336,280],[300,250],[300,300],[300,600]]' data-ad-keys="{}" data-ad-lazy="true" data-event-scroll-xs="" data-name="roba"></ins>
                </span></span></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">Los científicos organizaron un estudio en el que participaron 34 voluntarios con fibromialgia a las que se les asignó aleatoriamente el <strong>uso de varias gafas </strong>durante cuatro horas al día durante dos semanas. El reparto quedó de la siguiente manera:</p>
<p data-gtm-element-container="modulo-texto-link" class="paragraph"></p>
<ul><li>10 participantes llevaban gafas con <strong>cristales azules</strong>
</li></ul>
<p data-gtm-element-container="modulo-texto-link" class="paragraph"></p>
<ul>
<li>12 participantes llevaban gafas con <strong>cristales transparentes</strong>
</li>
<li>12 participantes llevaban gafas con<strong> cristales verdes</strong>
</li>
</ul>
<span class="content-ad-module ">
    
    
    
<span class="content-ad visible-xs roba-container  hidden-ad"><span class="ad-unit roba">
                <ins data-ad-type="roba" data-ad-device="xs" data-ad-sizes='["fluid",[336,280],[300,250],[300,300],[300,600]]' data-ad-keys="{}" data-ad-lazy="true" data-event-scroll-xs="" data-name="roba"></ins>
                </span></span></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">Tanto los voluntarios con<strong> gafas azules y transparentes no notaron ninguna reducción en su ansiedad</strong>. Pero los que llevaron las <strong>gafas verdes</strong> sí vieron buenos resultados: eran <strong>cuatro veces más propensos a tener niveles de ansiedad reducidos </strong>y, por lo tanto, no tuvieron la necesidad de tomar tantos opioides como los otros dos grupos. Los medicamentos en el caso de la fibromialgia se suelen tomar a modo preventivo y tener miedo al dolor conduce a un mayor uso de pastillas. </p>
<span class="content-ad-module ">
    
    
    
<span class="content-ad visible-xs roba-container  hidden-ad"><span class="ad-unit roba">
                <ins data-ad-type="roba" data-ad-device="xs" data-ad-sizes='["fluid",[336,280],[300,250],[300,300],[300,600]]' data-ad-keys="{}" data-ad-lazy="true" data-event-scroll-xs="" data-name="roba"></ins>
                </span></span></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">De hecho, los resultados de estas gafas fueron tan buenos que, tal y como cuentan los investigadores, los voluntarios preguntaron si, una vez finalizado el ensayo, <strong>podían seguir usando las gafas</strong>. Los investigadores también están estudiando si las gafas con cristales verdes se pueden recomendar a pacientes con otras afecciones de dolor crónico.</p>
<span class="content-subtitle-simple ">

    

<h2 class="subtitle">¿Cómo funcionan estas gafas especiales?</h2></span><p data-gtm-element-container="modulo-texto-link" class="paragraph">Padma Gulur, autora principal del estudio, ha explicado que las gafas con cristales verdes están<strong> especialmente diseñadas</strong> para filtrar una<strong> longitud de onda específica del espectro de la luz verde</strong> que<strong> estimula las vías en el cerebro </strong>que ayudan a <strong>controlar el dolor</strong>. Los investigadores creen que el alivio del dolor comienza "con la <strong>activación de los fotorreceptores del ojo </strong>que se comunican con partes del cerebro y modulan la respuesta al dolor mediante la activación de las vías inhibidoras del dolor descendentes".</p>
<p data-gtm-element-container="modulo-texto-link" class="paragraph">Estos hallazgos son especialmente esperanzadores porque hay una "necesidad urgente" de tratamientos adicionales par<strong>a reducir el uso de opioides entre los pacientes con fibromialgia</strong> y otros tipos de dolor crónico. "Las gafas verdes podrían proporcionar una <strong>opción fácil de usar y sin medicamento</strong>s", ha sentenciado la experta. </p>
