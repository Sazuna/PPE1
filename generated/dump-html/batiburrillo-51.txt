                            <p class="paragraph " id="m92-91-93"><strong>Duelo en el TC.</strong> De adversarios legítimos capaces de pactar por mayoría amplia asuntos relevantes, <a href="https://www.20minutos.es/minuteca/psoe/" title="PSOE">PSOE</a> y <a href="https://www.20minutos.es/minuteca/pp/" title="PP">PP</a> se han convertido en enemigos encarnizados. El fondo del asunto, <a href="https://www.20minutos.es/minuteca/cgpj/" title="CGPJ">el bloqueo a la renovación del Poder Judicial</a>, se ha visto superado por las formas: un batiburrillo de enmiendas planteadas apresuradamente por la vía de la reforma del Código Penal. El Constitucional afronta hoy recursos y recusaciones de PP, PSOE y UP. <strong>Es básico que se le deje deliberar sin ruido y con respeto.</strong></p></div>

                    : detail.path;

                debug(`Block ${blockId} (${url})`);

                const blockElement = document.querySelector(blockSelector);
                if (!blockElement) {
                    return debug(`Block ${blockSelector} not found`);
                }

                const taboolaWrapper = blockElement.querySelector(
                    '.taboola_container > div'
                );
                if (!taboolaWrapper) {
                    return debug(`Element ${taboolaWrapperSelector} not found`);
                }
                if (taboolaWrapper.hasChildNodes()) {
                    return debug(`Element ${taboolaWrapperSelector} already loaded`);
                }

                const blockIndex = blockId.split('_').pop();
                const elementId = `taboola-below-article-thumbnail_${blockIndex}`;
                taboolaWrapper.id = elementId;

                if (isMobile() && isFirstPage) {
                    debug('First page of mobile');
                    _taboola.push({
                        mode: 'thumbnails-a-1x1',
                        container: elementId,
                        placement: `Below Article Thumbnails ${blockIndex}`,
                        target_type: 'mix',
                    });

                    _taboola.push({
                        listenTo: 'render',
                        handler: function (e) {
                            let mode = e.detail.baseMode;
                            debug(`Handler ${mode} (${e.detail.placement})`);

                            switch (mode) {
                                case 'thumbnails-a-1x1-loyal':
                                    onTaboolaFinish(blockElement, 'loyal');
                                    break;
                                case 'thumbnails-a-1x1-casual':
                                    onTaboolaFinish(blockElement, 'casual');
                                    break;
                                case 'thumbnails-a-1x1-flyby':
                                    onTaboolaFinish(blockElement, 'flyby');
                                    break;
                            }
                        },
                    });
                } else {
                    debug('Desktop or next pages');

                    blockElement
                        .querySelectorAll('.WIDGET-PROMETEO-V2.prometeo-external')
                        .forEach((elem) => {
                            debug(`Remove unnecessary widget ${elem.classList.value}`);
                            elem.remove();
                        });

                    _taboola.push({
                        mode: 'thumbnails-a',
                        container: elementId,
                        placement: elementId,
                        target_type: 'mix',
                    });
                }

                _taboola.push({
                    article: 'auto',
                    url,
                });

                console.log('NewTBL: ' + message);

                debug('Taboola segment found: ' + userSegment);
                blockElement
                    .querySelectorAll('.WIDGET-PROMETEO-V2.prometeo-external')
                    .forEach((elem) => {
                        if (!elem.classList.contains(userSegment)) {
                            debug(`Remove unnecessary widget ${elem.classList.value}`);
                            elem.remove();
                        }
                    });

                        once: true,

            document.addEventListener('is.append', (e) => loadTaboola(false, e.detail));
