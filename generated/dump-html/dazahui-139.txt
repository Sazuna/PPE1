        function getParameter(paraStr, url)
{
    var result = "";
    //获取URL中全部参数列表数据
    var str = "&" + url.split("?")[1];
    var paraName = paraStr + "=";
    //判断要获取的参数是否存在
    if(str.indexOf("&"+paraName)!=-1)
    {
        //如果要获取的参数到结尾是否还包含“&”
        if(str.substring(str.indexOf(paraName),str.length).indexOf("&")!=-1)
        {
            //得到要获取的参数到结尾的字符串
            var TmpStr=str.substring(str.indexOf(paraName),str.length);
            //截取从参数开始到最近的“&”出现位置间的字符
            result=TmpStr.substr(TmpStr.indexOf(paraName),TmpStr.indexOf("&")-TmpStr.indexOf(paraName));  
        }
        else
        {  
            result=str.substring(str.indexOf(paraName),str.length);  
        }
    }  
    else

    {  
        result="无此参数";  
    }  
    result.replace("&","");
    return (result.split("=")[1]);  
    //return (result.replace("&",""));  
}

                        <div class="articleText"><p><strong><a href="http://www.cankaoxiaoxi.com?toindex=2" target="_blank">参考消息网</a>11月5日报道</strong> 据日本《朝日消息》11月4日报道，违章建筑导致道路变窄，是造成拥挤踩踏事故的原因之一。关于造成150多人死亡的韩国首尔繁华街区梨泰院的踩踏事故，韩国媒体和建筑专家提出了上述看法。在现场附近的小巷里，有多幢建筑显著向外凸出，可能阻碍了人流通过。</p><p>报道称，事故发生地周边狭窄的小巷里，是一间接一间的餐饮店。从店铺伸到小巷里的露台、舞台、长椅、空调外机等随处可见。建筑工程学专家、曾任韩国建国大学教授的安亨濬说：&ldquo;简直就是违章扩建的大杂烩。&rdquo;</p><p>事实上，那里的一些建筑被管辖梨泰院的龙山区认定违反《建筑法》，并处以&ldquo;强制罚款&rdquo;。但据安亨濬介绍，店家即使支付了强制罚款，只要继续经营，就会赚回超过罚款的利润，因此违规现象屡禁不止。</p><p>安亨濬就此次踩踏事故表示：&ldquo;由于这些扩建物挡住了部分道路，使人流受阻，在那些点位有可能发生了人群倒向一边的情况。&rdquo;</p><p>特别受到韩国媒体和专家关注的是，位于现场坡道东侧的汉密尔顿酒店建筑。它从地铁站旁一直延伸到坡道尽头的丁字路口。</p></div>
